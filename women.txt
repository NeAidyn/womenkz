    <!-- Скрипт для навигации и AI-чата -->
    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const navLinks = document.querySelectorAll('.nav-link');
            const pages = document.querySelectorAll('.page-content');
            const pageTitle = document.getElementById('page-title');

            // --- 1. Логика навигации ---
            function showPage(hash) {
                // Убираем #
                const pageId = hash.substring(1); 
                
                // Скрываем все страницы
                pages.forEach(page => {
                    page.classList.add('hidden');
                });
                
                // Показываем нужную страницу
                const activePage = document.getElementById(pageId) || document.getElementById('dashboard');
                activePage.classList.remove('hidden');

                // Обновляем активную ссылку
                navLinks.forEach(link => {
                    if (link.getAttribute('href') === hash) {
                        link.classList.add('bg-ebrd-blue-medium', 'text-white', 'border-l-4', 'border-ebrd-gold-light');
                        pageTitle.textContent = link.querySelector('span').textContent;
                    } else {
                        link.classList.remove('bg-ebrd-blue-medium', 'text-white', 'border-l-4', 'border-ebrd-gold-light');
                    }
                });
            }

            // Обработка кликов по навигации
            document.getElementById('main-nav').addEventListener('click', (e) => {
                let target = e.target.closest('a');
                if (target && target.classList.contains('nav-link')) {
                    e.preventDefault();
                    const hash = target.getAttribute('href');
                    window.location.hash = hash;
                }
            });
            
            // Обработка клика по профилю (бонус)
             document.querySelector('a[href="#profile"]').addEventListener('click', (e) => {
                 e.preventDefault();
                 window.location.hash = '#profile';
             });

            // Показать страницу при загрузке (на основе хеша или по умолчанию)
            showPage(window.location.hash || '#dashboard');
            
            // Слушатель изменения хеша (для кнопок "Назад/Вперед" в браузере)
            window.addEventListener('hashchange', () => {
                showPage(window.location.hash || '#dashboard');
            });


            // --- 2. Логика AI-чата (Gemini API) ---
            const sendBtn = document.getElementById('send-btn');
            const chatInput = document.getElementById('chat-input');
            const chatWindow = document.getElementById('chat-window');
            const loader = document.getElementById('loader');
            const sourcesContainer = document.getElementById('sources-container');
            const sourcesList = document.getElementById('sources-list');

            sendBtn.addEventListener('click', handleChatSubmit);
            chatInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    handleChatSubmit();
                }
            });

            function handleChatSubmit() {
                const userQuery = chatInput.value.trim();
                if (userQuery === "") return;

                // Добавляем сообщение пользователя в чат
                addMessageToChat(userQuery, 'user');
                chatInput.value = "";
                loader.classList.remove('hidden');
                sourcesContainer.classList.add('hidden');
                sourcesList.innerHTML = '';

                // Вызываем API
                callGeminiAPI(userQuery);
            }

            function addMessageToChat(message, sender) {
                const messageEl = document.createElement('div');
                if (sender === 'user') {
                    messageEl.className = 'flex justify-end';
                    messageEl.innerHTML = `
                        <div class="bg-ebrd-blue-dark text-white p-3 rounded-lg max-w-lg">
                            <p class="text-sm">${message}</p>
                        </div>
                    `;
                } else {
                    messageEl.className = 'flex items-start space-x-3';
                    messageEl.innerHTML = `
                        <div class="w-10 h-10 rounded-full bg-ebrd-blue-dark text-white flex items-center justify-center flex-shrink-0">
                            <i class="fas fa-robot"></i>
                        </div>
                        <div class="bg-gray-100 p-3 rounded-lg max-w-lg">
                            <!-- Текст будет вставлен после обработки markdown -->
                        </div>
                    `;
                    // Простой Markdown парсер (для **bold**)
                    let htmlMessage = message
                        .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                        .replace(/\*(.*?)\*/g, '<em>$1</em>')
                        .replace(/\n/g, '<br>');
                    
                    messageEl.querySelector('.bg-gray-100').innerHTML = `<p class="text-sm">${htmlMessage}</p>`;
                }
                chatWindow.appendChild(messageEl);
                // Прокрутка вниз
                chatWindow.scrollTop = chatWindow.scrollHeight;
            }

            // Функция для вызова Gemini API
            async function callGeminiAPI(userQuery) {
                // Системная инструкция, основанная на целях программы (из PDF)
                const systemPrompt = `Ты - "AI Бизнес-Консультант" для программы ЕБРР "Менторинг для женщин-предпринимателей Казахстана". 
Твоя цель - помогать участницам повышать компетенции в управлении бизнесом, маркетинге, финансах и инновациях. 
Отвечай профессионально, дружелюбно и по существу. 
Всегда предоставляй конкретные, действенные советы. 
Ты должен использовать поиск Google (grounding) для предоставления самой актуальной информации, особенно по трендам, инструментам или рыночным данным.`;

                const apiKey = ""; // API-ключ не требуется, будет предоставлен средой выполнения
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

                const payload = {
                    contents: [{
                        parts: [{ text: userQuery }]
                    }],
                    // Включаем Google Search (grounding) для получения актуальной информации
                    tools: [{ "google_search": {} }],
                    systemInstruction: {
                        parts: [{ text: systemPrompt }]
                    },
                };

                try {
                    // Используем экспоненциальную задержку
                    let response;
                    let retries = 3;
                    let delay = 1000;

                    for (let i = 0; i < retries; i++) {
                        response = await fetch(apiUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify(payload)
                        });

                        if (response.ok) {
                            break; // Успех, выходим из цикла
                        }

                        if (!response.ok && response.status === 429) {
                            // Слишком много запросов, ждем и пробуем снова
                            await new Promise(resolve => setTimeout(resolve, delay));
                            delay *= 2; // Увеличиваем задержку
                        } else {
                            // Другая ошибка
                            throw new Error(`HTTP error! status: ${response.status}`);
                        }
                    }
                    
                    if (!response.ok) {
                         throw new Error(`Failed to get response after ${retries} retries.`);
                    }

                    const result = await response.json();
                    
                    loader.classList.add('hidden');

                    const candidate = result.candidates?.[0];
                    if (candidate && candidate.content?.parts?.[0]?.text) {
                        const text = candidate.content.parts[0].text;
                        addMessageToChat(text, 'ai');

                        // Обработка источников (grounding metadata)
                        const groundingMetadata = candidate.groundingMetadata;
                        if (groundingMetadata && groundingMetadata.groundingAttributions) {
                            const sources = groundingMetadata.groundingAttributions
                                .map(attribution => ({
                                    uri: attribution.web?.uri,
                                    title: attribution.web?.title,
                                }))
                                .filter(source => source.uri && source.title); // Убедимся, что источники валидны

                            if (sources.length > 0) {
                                sources.forEach(source => {
                                    const li = document.createElement('li');
                                    li.innerHTML = `<a href="${source.uri}" target="_blank" class="text-ebrd-blue-dark hover:underline">${source.title}</a>`;
                                    sourcesList.appendChild(li);
                                });
                                sourcesContainer.classList.remove('hidden');
                            }
                        }
                    } else {
                        addMessageToChat("Извините, я не смог обработать ваш запрос. Попробуйте переформулировать.", 'ai');
                    }
                } catch (error) {
                    console.error("Ошибка при вызове Gemini API:", error);
                    loader.classList.add('hidden');
                    addMessageToChat("Произошла ошибка при подключении к AI-консультанту. Пожалуйста, попробуйте позже.", 'ai');
                }
            }
        });
    </script>
